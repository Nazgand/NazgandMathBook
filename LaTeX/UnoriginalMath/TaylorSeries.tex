	When the series converges,
	\begin{equation}
		\label{TaylorSeriesDef}
		f\pqty{x}=\sum_{k=0}^{\infty}\frac{\pqty{x-a}^k\pqty{\pdiff{z}{k}f\pqty{z} \replace{z}{a}}}{k!}
		=\sum_{k=0}^{\infty}\frac{\pqty{x-a}^k f^{\pqty{k}}\pqty{a}}{k!}
	\end{equation}
