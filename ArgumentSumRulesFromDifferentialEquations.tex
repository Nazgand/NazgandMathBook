	\begin{equation}
		f\pqty{z}=f''\pqty{z}
		\Rightarrow
		f\pqty{a+b}=
		\frac
		{f\pqty{a}\pqty{f\pqty{0}f\pqty{b}-f'\pqty{0}f'\pqty{b}}
		+f'\pqty{a}\pqty{f\pqty{0}f'\pqty{b}-f'\pqty{0}f\pqty{b}}
		}
		{f\pqty{0}^2-f'\pqty{0}^2}
	\end{equation}
