	When the series converges,
	\begin{equation}
		\label{TaylorSeriesDef}
		f\pqty{x}=\sum_{k=0}^{\infty}\frac{\pqty{x-a}^k\pqty{\pdiff{z}{k}f\pqty{z}:z\to a}}{k!}
	\end{equation}
